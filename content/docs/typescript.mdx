# TypeScript in Elide

Elide provides first-class TypeScript support with automatic compilation, type checking, and full IDE integration.

## Features

- **Zero Configuration**: No build step required
- **Automatic Compilation**: TypeScript compiles on the fly
- **Type Checking**: Full TypeScript type system support
- **Source Maps**: Debug TypeScript directly
- **Declaration Files**: Generate `.d.ts` files

## Getting Started

Simply write TypeScript—Elide handles the rest:

```typescript
// app.ts
interface User {
  id: number
  name: string
  email: string
  createdAt: Date
}

async function createUser(data: Omit<User, 'id' | 'createdAt'>): Promise<User> {
  const user: User = {
    ...data,
    id: Math.random(),
    createdAt: new Date()
  }
  
  return user
}

// Run directly
const newUser = await createUser({
  name: 'Alice',
  email: 'alice@example.com'
})

console.log(newUser)
```

Run it:

```bash
elide run app.ts
```

No build step needed!

## Type Safety

### Interface Definitions

```typescript
interface ApiResponse<T> {
  success: boolean
  data?: T
  error?: string
  timestamp: number
}

interface User {
  id: number
  name: string
  email: string
}

async function fetchUser(id: number): Promise<ApiResponse<User>> {
  const response = await fetch(`/api/users/${id}`)
  return response.json()
}

const result = await fetchUser(123)
if (result.success && result.data) {
  console.log(result.data.name) // Type-safe!
}
```

### Type Guards

```typescript
function isString(value: unknown): value is string {
  return typeof value === 'string'
}

function processValue(value: unknown) {
  if (isString(value)) {
    // TypeScript knows value is string here
    console.log(value.toUpperCase())
  }
}

// Discriminated unions
type Result<T> = 
  | { success: true; value: T }
  | { success: false; error: string }

function handleResult<T>(result: Result<T>): T {
  if (result.success) {
    return result.value // Type-safe access
  } else {
    throw new Error(result.error)
  }
}
```

### Generics

```typescript
// Generic function
function first<T>(arr: T[]): T | undefined {
  return arr[0]
}

const nums = [1, 2, 3]
const firstNum = first(nums) // type: number | undefined

// Generic class
class Cache<T> {
  private data = new Map<string, T>()
  
  set(key: string, value: T): void {
    this.data.set(key, value)
  }
  
  get(key: string): T | undefined {
    return this.data.get(key)
  }
}

const userCache = new Cache<User>()
userCache.set('123', { id: 123, name: 'Alice', email: 'alice@example.com' })
```

## Elide API Types

### HTTP Server Types

```typescript
import type { Request, Response, RouteHandler } from '@elide/server'

const handler: RouteHandler = (req: Request, res: Response) => {
  const userId = req.params.id
  
  res.json({
    id: userId,
    name: 'Alice'
  })
}
```

### File System Types

```typescript
import type { FileStats, FileHandle } from '@elide/fs'

async function getFileInfo(path: string): Promise<FileStats> {
  return await Elide.fs.stat(path)
}

const stats = await getFileInfo('./file.txt')
console.log(stats.size, stats.mtime)
```

## Configuration

### tsconfig.json

Create a `tsconfig.json` for IDE support:

```json
{
  "compilerOptions": {
    "target": "ES2024",
    "module": "ESNext",
    "lib": ["ES2024"],
    "moduleResolution": "bundler",
    "strict": true,
    "esModuleInterop": true,
    "skipLibCheck": true,
    "forceConsistentCasingInFileNames": true,
    "types": ["@elide/types"]
  },
  "include": ["src/**/*"],
  "exclude": ["node_modules"]
}
```

## Decorators

Experimental decorator support:

```typescript
// Enable decorators in tsconfig.json
{
  "compilerOptions": {
    "experimentalDecorators": true,
    "emitDecoratorMetadata": true
  }
}

// Use decorators
function Route(path: string) {
  return function (target: any, key: string) {
    target.routes = target.routes || []
    target.routes.push({ path, handler: key })
  }
}

class API {
  @Route('/users')
  async getUsers() {
    return [{ id: 1, name: 'Alice' }]
  }
  
  @Route('/users/:id')
  async getUser(id: number) {
    return { id, name: 'Alice' }
  }
}
```
